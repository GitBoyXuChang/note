过滤

```java
    /**
     * 判断集合--过滤集合数据--去除重复的
     * @param empList
     * @return
     */
    private  List<AttendanceCode> doFilterEmpRecord(List<AttendanceCode> empList){
        for (int i = 0; i < empList.size()-1; i++) {
            for (int j = empList.size()-1; j > i; j--) {
                if (empList.get(j).getCode().equals(empList.get(i).getCode())) {
                    empList.remove(j);
                }
            }
        }
        return  empList;
    }
```

Collections.sort(recordresults, new Comparator<OapiAttendanceListResponse.Recordresult>() {
            @Override
            public int compare(OapiAttendanceListResponse.Recordresult o1, OapiAttendanceListResponse.Recordresult o2) {
                //条件优先级
                Date workdate1 = o1.getWorkDate();
                Date workdate2 = o2.getWorkDate();

                String userid1 = o1.getUserId();
                String userid2 = o2.getUserId();

                Date userCheckTime1 = o1.getUserCheckTime();
                Date userCheckTime2 = o2.getUserCheckTime();

                if(workdate1.getTime() == workdate2.getTime()){
                    if(userid1.equals(userid2)){
                        return userCheckTime1.compareTo(userCheckTime2);
                    }
                    return userid1.compareTo(userid2);
                }

                return workdate1.compareTo(workdate2);
            }
        });
